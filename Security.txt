Аутентификация - Проверка подлинности пользователя или процесса login-password
Авторизация - Процесс наделения правами и определение, что пользователь может делать после успешной аутентификации.

Архитектура

  Запрос
    |
  фильтр аутентификации (authentication filter)                 контекст безопасности (security context) сохраняет данные
  делегирует запрос менеджеру аутентификации                --- аутентификации после процесса аутентификации. Контекст безопасности 
  и на основе ответа настраивает контекст безопасности          будет хранить данные до тех пор, пока действие не завершится
    |   |
  менеджер аутентификации (authentication manager) 
  использует провайдера аутентификации для обработки аутентификации
    |   |
  провайдер аутентификации (authentication provider)            ---   кодировщик паролей (password encoder) реализует управление паролями, 
  реализует логику аутентификации                                     которое провайдер аутентификации использует в логике аутентификации
  находит пользователя с помощью сервиса сведений о                   После нахождения данных пользователя AuthenticationProvider
  пользователе и проверяет пароль с помощью кодировщика паролей       проверяет пароль с помощью PasswordEncoder
    | 
  служба сведений о пользователе (user details service) 
  реализует ответственность за управление пользователями, 
  которую провайдер аутентификации использует в логике аутентификации
  
При использовании Spring Boot производится автонастройка по умолчанию
  
  защищаются все Endpoint
  
  HTTP Basic - аутентифицирует пользователя с помощью набора учетных данных (имя пользователя и пароль), 
              которые приложение получает в заголовке HTTP-запроса
  
  интерфейс UserDetailsService - хранение в памяти, создает пользователя user и паролем UUID выводимый в консоль
  интерфейс PasswordEncoder - отвечает за кодирование и сравнение
  
  данные интерфейсы обязательны и связаны, при реализации UserDetailsService необходимо реализовать PasswordEncoder

        CharSequence rawPassword, // пароль который мы получили с веба
        String encodedPassword // пароль полученный из UserDetailsService
        );
    // включает повторное кодирование закодированного пароля
    default boolean upgradeEncoding(@String encodedPassword)

  @Component
  public class CustomPasswordEncoder implements PasswordEncoder {

      @Override
      public @Nullable String encode(@Nullable CharSequence rawPassword) {
          return hashWithSHA512(rawPassword.toString());
      }

      @Override
      public boolean matches(@Nullable CharSequence rawPassword, @Nullable String encodedPassword) {
          String hashedPassword = encode(rawPassword);
          return encodedPassword.equals(hashedPassword);
      }

  Готовые реализации

  public class CustomFilter implements Filter {
      @Override
      public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain) throws IOException, ServletException {
          var httpRequest = (HttpServletRequest) request;
          var httpResponse = (HttpServletResponse) response;
          String requestId = httpRequest.getHeader("Request-Id");
          if (requestId == null || requestId.isBlank()) {
              httpResponse.setStatus(HttpServletResponse.SC_BAD_REQUEST);
              return;
          }
          chain.doFilter(request, response);
      }
  }

  @Component
  public class CustomFilter extends OncePerRequestFilter {
      
      @Value("${authorization.key}")
      private String authorizationKey;

      @Override
      protected void doFilterInternal(HttpServletRequest request, HttpServletResponse response, FilterChain filterChain) throws ServletException, IOException {

          String authentication = request.getHeader("Authorization");
          if (authorizationKey.equals(authentication))
              filterChain.doFilter(request, response);
          else
              response.setStatus(HttpServletResponse.SC_UNAUTHORIZED);

      }
  }

AuthenticationProvider
  под дефолту использует UserDetailsService и PasswordEncoder

  public @Nullable Authentication authenticate(Authentication authentication) throws AuthenticationException
  
    метод должен выдавать исключение AuthenticationException, если аутентификация не удалась
    если метод получает объект аутентификации, не поддерживаемый вашей реализацией AuthenticationProvider, то он должен возвращать null
    метод должен возвращать экземпляр Authentication, представляющий полностью аутентифицированный объект

  public boolean supports(Class<?> authentication)
    проверка поддержки типа объекта Authentication

  
  Может быть множество реализаций AuthenticationProvider поддерживающих разные типы Authentication
  Например один по логину паролю, другой по СМС коду, третий по ключу
  Причем можно сделать 2 разных по коду, для таких supports будет возвращать true, а authenticate будет возвращать null для неподдерживаемого
  
  @Component
  @RequiredArgsConstructor
  public class CusmonAuthenticationProvider implements AuthenticationProvider {

      private final UserDetailsService userDetailsService;
      private final PasswordEncoder passwordEncoder;

      @Override
      public @Nullable Authentication authenticate(Authentication authentication) throws AuthenticationException {

          String username = authentication.getName();
          String password = authentication.getCredentials().toString();

          UserDetails user = userDetailsService.loadUserByUsername(username);

          if (passwordEncoder.matches(password, user.getPassword())) {
              return new UsernamePasswordAuthenticationToken(user, password, user.getAuthorities());
          } else {
              throw new BadCredentialsException("Bad credentials");
          }
      }

      @Override
      public boolean supports(Class<?> authentication) {
          return authentication.equals(UsernamePasswordAuthenticationToken.class);
      }
  }

  фильтр CsrfFilter
    перехватывает запросы и пропускает только те из них, которые используют методы HTTP
    GET, HEAD, TRACE и OPTIONS. Для всех остальных запросов фильтр ожидает получить заголовок, содержащий токен. 
    Если этот заголовок не существует или содержит неверное значение токена, 
    приложение отклоняет запрос и возвращает статус ответа HTTP 403 Forbidden.

CORS
  
  В браузерах встроен механизм безопасности при открытии страницы, 
  если на ней происходит запрос к другому домену например через iframe, то он будет заблокирован
  
  Механизм CORS помогает ослабить эти ограничения
  работает на основе заголовков HTTP
  Браузер отправляет запрос с заголовком Origin. Сервер в ответ посылает Access-Control-Allow-Origin, разрешающий доступ, если источник в белом списке
  
  Access-Control-Allow-Origin: (например, * или https://example.com) кто может читать данные.
  Access-Control-Allow-Methods: разрешенные методы (GET, POST).
  Access-Control-Allow-Headers: разрешенные заголовки. 
  
  Применение
  
    через аннотацию в контроллере
      @CrossOrigin({"example.com", "example.org"}) - метод, делает конечную точку разрешенной, в str указываются хосты (также можно *), с которых можно ее вызвать
                                                    также в ней можно задать allowedHeaders и methods

    через конфиг
      
      http.cors(c -> {
          CorsConfigurationSource source = request -> {
              CorsConfiguration config = new CorsConfiguration();
              config.setAllowedOrigins(List.of("example.com", "example.org"));
              config.setAllowedMethods(List.of("GET", "POST", "PUT", "DELETE"));
              config.setAllowedHeaders(List.of("*"));
              return config;
          };
          c.configurationSource(source);
      });

    Реализация в отдельном классе для более сложной логики
      
      hasPermission в SpEL (Spring Expression Language) — это специальная функция Spring Security для детальной авторизации, 
          используемая с ACL (Access Control List) для проверки прав на конкретные объекты данных. 
          Она делегирует проверку компоненту PermissionEvaluator, позволяя разрешить или запретить доступ (READ, WRITE и т.д.) к методу или объекту.

      @GetMapping("/user/{id}")
      public User findById(@PathVariable("id") Long id) {...}
      
      реализуем контракт PermissionEvaluator
            Authentication authentication - биндится автоматически
        
        //Проверяет право доступа (permission) к уже существующему объекту (targetDomainObject)
        boolean hasPermission(Authentication authentication, Object targetDomainObject, Object permission);
        //Проверяет право доступа по ID и типу объекта, не загружая сам объект, например
        boolean hasPermission(Authentication authentication, Serializable targetId, String targetType, Object permission);
      
        @Component
        public class CustomPermissionEvaluator implements PermissionEvaluator {

            @Override
            public boolean hasPermission(Authentication authentication, Object targetDomainObject, Object permission) {
                User user = (User) targetDomainObject;
                String p = (String) permission;
                return authentication.getAuthorities().stream()
                        .anyMatch(a -> a.getAuthority().equals(p))
            }

            @Override
            public boolean hasPermission(Authentication authentication, Serializable targetId, String targetType, Object permission) {
                return false;
            }
        }

OAuth 2 и OpenID Connect 

  OAuth 2 - спецификация, которая описывает разделение обязанностей по аутентификации в системе
    
    Архитектура
      пользователь – субъект, который использует приложение
      клиент  – приложение, которое вызывает бэкенд и требует аутентификации и  авторизации
      сервер авторизации – приложение, которое реализует аутентификацию и безопасное хранение учетных данных
      сервер ресурсов – бэкенд-приложение, которое авторизует и обслуживает вызовы, отправленные одним или несколькими клиентскими приложениями
    
      клиент запрашивает токен у сервера авторизации - далее использует этот токен для доступа к серверу ресурсов
      
      Токены
        непрозрачные – токены, которые не хранят данные. Для реализации авторизации серверу ресурсов обычно необходимо вызвать сервер авторизации, 
                      предоставить непрозрачный токен и  получить данные. Это так называемый вызов интроспекции (introspection call)
        прозрачные  – токены, которые хранят данные, что позволяет бэкенду немедленно реализовать авторизацию. Наиболее часто
                      используемой реализацией прозрачного токена является JSON Web Token (JWT)
      
        JSON Web Token
          заголовок – обычно содержит данные о токене, такие как криптографический алгоритм, используемый для подписи токена,
                      или идентификатор ключа, который сервер авторизации использовал для его подписи;
          тело – обычно содержит данные о субъекте, которому был выдан токен, такие как данные клиента и пользователя;
          подпись – криптографически сгенерированное значение, которое можно использовать для доказательства того, 
                    что сервер авторизации действительно выдал токен и что никто не изменил его содержимое (заголовок или тело) после его генерации.
    
    
        Получение токена (grant type)
          Грант в форме кода авторизации
              при успешной аутентификации сервер авторизации выдает клинту код авторизации,
              затем по коду авторизации клиент получает у сервера авторизации токен
              
          Защита PKCE совместно с кодом авторизации 
              это защита шага получения токена
          
              клиент генерирует верификатор (verifier) - случайное значение, применяет к нему хеш функцию и получает "вызов" (challenge)
              отправляет его вместо с логином и паролем, сервер авторизации сохраняет его
              при запросе токена дополнительно к ключу передается challenge и проверяется сервером авторизации
          
          Получение токена с предоставлением учетных данных клиента (например когда клиента нет, когда службе необходимо вызвать другую службу)
              Приложение запрашивает токен доступа с сервера авторизации. Приложение использует свои учетные данные для аутентификации.
              Если учетные данные действительны, сервер авторизации выдает токен доступа.
              Приложение использует токен доступа для авторизации при отправке запроса на сервер ресурсов.
          
          Использование обновления токенов для получения новых токенов доступа
              Клиент отправляет токен обновления, чтобы доказать, что он уже прошел аутентификацию ранее.
              Сервер авторизации распознает токен обновления и  предоставляет клиенту новый токен доступа.

                  // идентификатор клиента – внешний идентификатор клиента, аналогичный имени пользователя

                  // секрет клиента – аналог пароля пользователя

                  // Область действия можно использовать позже в правилах авторизации
                  .scope(OidcScopes.OPENID)
                  .build();

      }

      import com.nimbusds.jose.jwk.RSAKey;
      import com.nimbusds.jose.proc.SecurityContext;
      
      @Bean
      public JWKSource<SecurityContext> jwkSource() throws NoSuchAlgorithmException {
          KeyPairGenerator keyPairGenerator = KeyPairGenerator.getInstance("RSA");
          keyPairGenerator.initialize(2048);
          KeyPair keyPair = keyPairGenerator.generateKeyPair();
          RSAPublicKey publicKey = (RSAPublicKey) keyPair.getPublic();
          RSAPrivateKey privateKey = (RSAPrivateKey) keyPair.getPrivate();
          RSAKey rsaKey = new RSAKey.Builder(publicKey)
                       .privateKey(privateKey)
                       .keyID(UUID.randomUUID().toString())
                       .build();
          JWKSet jwkSet = new JWKSet(rsaKey);
          return new ImmutableJWKSet<>(jwkSet);
      }
